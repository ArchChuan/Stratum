# Memory v2 End-to-End Testing Plan (Phase 10)

**Goal:** Build E2E test suite covering full memory lifecycle: message → extraction → recall → forget → GC, plus admin diagnostics and Agent integration.

**Architecture:** Go integration tests using real PostgreSQL + Redis + Milvus (testcontainers), mocked LLM responses. 8 test scenarios covering happy path + edge cases + error recovery. Each test runs in isolated tenant schema.

**Tech Stack:** Go 1.22+, testify/suite, testcontainers-go, httptest

---

## Global Constraints

- All tests use testcontainers for PostgreSQL/Redis/Milvus (no docker-compose required)
- Each test creates isolated tenant schema (cleanup via DROP SCHEMA CASCADE)
- LLM responses mocked with deterministic JSON (no real API calls)
- Tests run sequentially (not parallel) to avoid resource contention
- Cleanup: DROP tenant schema + Milvus collection after each test
- CI timeout: 10 minutes per test, 60 minutes total suite
- Test coverage target: ≥80% for critical paths

```go
// test/e2e/testutil.go
package e2e_test

import (
 "context"
 "fmt"
 "testing"
 "time"

 "github.com/jackc/pgx/v5/pgxpool"
 "github.com/redis/go-redis/v9"
 "github.com/stretchr/testify/require"
 "github.com/testcontainers/testcontainers-go"
 "github.com/testcontainers/testcontainers-go/wait"

 "github.com/byteBuilderX/stratum/api/wiring"
 "github.com/byteBuilderX/stratum/internal/memory/application"
)

type TestEnv struct {
 PGPool      *pgxpool.Pool
 Redis       *redis.Client
 MemoryService *application.MemoryService
 TenantID    string
 UserID      string
 AgentID     string
 Cleanup     func()
}

func SetupMemoryTestEnv(t *testing.T) *TestEnv {
 t.Helper()
 ctx := context.Background()

 // Start PostgreSQL
 pgContainer, err := testcontainers.GenericContainer(ctx, testcontainers.GenericContainerRequest{
  ContainerRequest: testcontainers.ContainerRequest{
   Image:        "postgres:15-alpine",
   ExposedPorts: []string{"5432/tcp"},
   Env: map[string]string{
    "POSTGRES_PASSWORD": "test",
    "POSTGRES_DB":       "stratum_test",
   },
   WaitingFor: wait.ForLog("database system is ready to accept connections"),
  },
  Started: true,
 })
 require.NoError(t, err)

 pgHost, err := pgContainer.Host(ctx)
 require.NoError(t, err)
 pgPort, err := pgContainer.MappedPort(ctx, "5432")
 require.NoError(t, err)

 pgDSN := fmt.Sprintf("postgres://postgres:test@%s:%s/stratum_test?sslmode=disable", pgHost, pgPort.Port())
 pgPool, err := pgxpool.New(ctx, pgDSN)
 require.NoError(t, err)

 // Start Redis
 redisContainer, err := testcontainers.GenericContainer(ctx, testcontainers.GenericContainerRequest{
  ContainerRequest: testcontainers.ContainerRequest{
   Image:        "redis:7-alpine",
   ExposedPorts: []string{"6379/tcp"},
   WaitingFor:   wait.ForLog("Ready to accept connections"),
  },
  Started: true,
 })
 require.NoError(t, err)

 redisHost, err := redisContainer.Host(ctx)
 require.NoError(t, err)
 redisPort, err := redisContainer.MappedPort(ctx, "6379")
 require.NoError(t, err)

 redisClient := redis.NewClient(&redis.Options{
  Addr: fmt.Sprintf("%s:%s", redisHost, redisPort.Port()),
 })

 // Create tenant schema
 tenantID := fmt.Sprintf("test_%d", time.Now().UnixNano())
 schemaName := fmt.Sprintf("tenant_%s", tenantID)
 _, err = pgPool.Exec(ctx, fmt.Sprintf("CREATE SCHEMA %s", schemaName))
 require.NoError(t, err)

 // Apply tenant DDL
 tenantDDL := loadTenantDDL(t)
 _, err = pgPool.Exec(ctx, fmt.Sprintf("SET search_path = %s, public; %s", schemaName, tenantDDL))
 require.NoError(t, err)

 // Build MemoryService with mocked LLM
 memoryService := buildMemoryService(t, pgPool, redisClient)

 cleanup := func() {
  _, _ = pgPool.Exec(ctx, fmt.Sprintf("DROP SCHEMA %s CASCADE", schemaName))
  pgPool.Close()
  redisClient.Close()
  _ = pgContainer.Terminate(ctx)
  _ = redisContainer.Terminate(ctx)
 }

 return &TestEnv{
  PGPool:        pgPool,
  Redis:         redisClient,
  MemoryService: memoryService,
  TenantID:      tenantID,
  UserID:        "test_user",
  AgentID:       "test_agent",
  Cleanup:       cleanup,
 }
}

func loadTenantDDL(t *testing.T) string {
 // Load from pkg/storage/postgres/tenant_schema.sql
 return `CREATE TABLE memory_facts (...); CREATE TABLE memory_entities (...);`
}

func buildMemoryService(t *testing.T, pool *pgxpool.Pool, redis *redis.Client) *application.MemoryService {
 // Wire MemoryService with mocked LLM extractor
 return application.NewMemoryService(
  persistence.NewFactRepo(pool),
  persistence.NewEntityRepo(pool),
  persistence.NewExtractionQueue(pool),
  &mockVectorStore{},
  &mockLLMExtractor{},
  &mockEmbedClient{},
  redis,
 )
}

type mockLLMExtractor struct{}

func (m *mockLLMExtractor) ExtractFacts(ctx context.Context, messages []port.Message) (*port.ExtractionResult, error) {
 return &port.ExtractionResult{
  Facts: []port.ExtractedFact{
   {
    Content:    "User prefers dark mode",
    Importance: 0.8,
    Keywords:   []string{"preference", "ui"},
    Entities: []port.ExtractedEntity{
     {Name: "dark mode", Type: "preference"},
    },
   },
  },
 }, nil
}

type mockEmbedClient struct{}

func (m *mockEmbedClient) EmbedText(ctx context.Context, text string) ([]float32, error) {
 vec := make([]float32, 1024)
 for i := range vec {
  vec[i] = 0.1
 }
 return vec, nil
}

func (m *mockEmbedClient) Dimension() int { return 1024 }

type mockVectorStore struct {
 facts map[string][]float32
}

func (m *mockVectorStore) UpsertFact(ctx context.Context, tenantID, factID string, vector []float32, metadata map[string]interface{}) error {
 if m.facts == nil {
  m.facts = make(map[string][]float32)
 }
 m.facts[factID] = vector
 return nil
}

func (m *mockVectorStore) SearchFacts(ctx context.Context, tenantID string, queryVector []float32, scopeExpr string, topK int) ([]*port.VectorHit, error) {
 var hits []*port.VectorHit
 for id := range m.facts {
  hits = append(hits, &port.VectorHit{ID: id, Score: 0.95, Metadata: map[string]interface{}{}})
 }
 return hits, nil
}

func (m *mockVectorStore) DeleteFact(ctx context.Context, tenantID, factID string) error {
 delete(m.facts, factID)
 return nil
}
```

```go
// test/e2e/memory_lifecycle_test.go
package e2e_test

import (
 "context"
 "testing"
 "time"

 "github.com/byteBuilderX/stratum/internal/memory/application"
 "github.com/stretchr/testify/require"
)

func TestMemoryLifecycle_HappyPath(t *testing.T) {
 env := SetupMemoryTestEnv(t)
 defer env.Cleanup()

 ctx := context.Background()

 // Step 1: Buffer 5 messages (triggers flush)
 for i := 0; i < 5; i++ {
  req := &application.BufferMessageRequest{
   TenantID:       env.TenantID,
   UserID:         env.UserID,
   AgentID:        env.AgentID,
   ConversationID: "conv001",
   MessageID:      fmt.Sprintf("msg%03d", i),
   Role:           "user",
   Content:        fmt.Sprintf("Message %d: I prefer dark mode", i),
   CreatedAt:      time.Now(),
  }
  err := env.MemoryService.BufferMessage(ctx, req)
  require.NoError(t, err)
 }

 // Step 2: Wait for extraction (queue should have 1 item)
 time.Sleep(500 * time.Millisecond)

 // Step 3: Poll and extract facts
 extractReq := &application.ExtractFactsRequest{
  TenantID:       env.TenantID,
  UserID:         env.UserID,
  AgentID:        env.AgentID,
  ConversationID: "conv001",
  Messages: []application.Message{
   {Role: "user", Content: "I prefer dark mode"},
  },
 }
 err := env.MemoryService.ExtractFacts(ctx, extractReq)
 require.NoError(t, err)

 // Step 4: Recall memory
 recallReq := &application.RecallMemoryRequest{
  TenantID:  env.TenantID,
  UserID:    env.UserID,
  AgentID:   env.AgentID,
  ReadScope: "user",
  Query:     "user preferences",
  TopK:      10,
 }
 recallResp, err := env.MemoryService.RecallMemory(ctx, recallReq)
 require.NoError(t, err)
 require.NotEmpty(t, recallResp.Facts, "should have extracted facts")
 require.Contains(t, recallResp.Facts[0].Content, "dark mode")

 // Step 5: Forget fact
 factID := recallResp.Facts[0].ID
 forgetReq := &application.ForgetMemoryRequest{
  TenantID: env.TenantID,
  UserID:   env.UserID,
  FactID:   factID,
 }
 err = env.MemoryService.ForgetMemory(ctx, forgetReq)
 require.NoError(t, err)

 // Step 6: Verify fact no longer recalled
 recallResp2, err := env.MemoryService.RecallMemory(ctx, recallReq)
 require.NoError(t, err)
 for _, fact := range recallResp2.Facts {
  require.NotEqual(t, factID, fact.ID, "forgotten fact should not appear")
 }
}
```

```go
// test/e2e/memory_context_test.go
package e2e_test

import (
 "context"
 "testing"

 "github.com/byteBuilderX/stratum/internal/agent/application"
 "github.com/byteBuilderX/stratum/internal/memory/application"
 "github.com/stretchr/testify/require"
)

func TestMemoryContext_InjectedIntoAgent(t *testing.T) {
 env := SetupMemoryTestEnv(t)
 defer env.Cleanup()

 ctx := context.Background()

 // Step 1: Insert a fact
 extractReq := &application.ExtractFactsRequest{
  TenantID: env.TenantID,
  UserID:   env.UserID,
  AgentID:  env.AgentID,
  Messages: []application.Message{
   {Role: "user", Content: "I prefer Python over JavaScript"},
  },
 }
 err := env.MemoryService.ExtractFacts(ctx, extractReq)
 require.NoError(t, err)

 // Step 2: Build context
 buildReq := &application.BuildContextRequest{
  TenantID:  env.TenantID,
  UserID:    env.UserID,
  AgentID:   env.AgentID,
  ReadScope: "user",
  TopK:      10,
 }
 buildResp, err := env.MemoryService.BuildContext(ctx, buildReq)
 require.NoError(t, err)
 require.NotEmpty(t, buildResp.ContextText, "should have context text")
 require.Contains(t, buildResp.ContextText, "Python")

 // Step 3: Verify entity profile
 require.NotEmpty(t, buildResp.EntityProfiles, "should have entity profiles")
 pythonEntity := findEntity(buildResp.EntityProfiles, "Python")
 require.NotNil(t, pythonEntity, "should have Python entity")
 require.Equal(t, "technology", pythonEntity.Type)
}

func findEntity(profiles []application.EntityProfileDTO, name string) *application.EntityProfileDTO {
 for i := range profiles {
  if profiles[i].Name == name {
   return &profiles[i]
  }
 }
 return nil
}
```
