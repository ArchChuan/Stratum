> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建 `mcp-delve`，一个封装 Go delve DAP 协议的 Node.js MCP server，让 Claude 能通过工具调用真正控制 Go 程序的断点调试。

**Architecture:** Node.js 进程通过 stdio 与 Claude 通信（MCP 协议）；调试会话启动时 spawn `dlv dap --listen=:<port>` 子进程，服务器再通过 TCP 连接 dlv 并说 DAP 协议。所有调试状态（线程 ID、帧 ID、序列号）维护在 DebugSession 单例中，工具调用之间持久化。

**Tech Stack:** Node.js v24，`@modelcontextprotocol/sdk`，`zod`，原生 `net` 模块（TCP DAP 通信），`dlv dap`（delve v1.26.3）

写文件 `~/.claude/mcp-servers/mcp-delve/src/dap-client.js`：

```js
import { EventEmitter } from 'events';
import net from 'net';

export class DAPClient extends EventEmitter {
  constructor() {
    super();
    this._seq = 1;
    this._pending = new Map();   // seq → { resolve, reject }
    this._buf = '';
    this._socket = null;
  }

  connect(host, port) {
    return new Promise((resolve, reject) => {
      const sock = net.createConnection({ host, port }, () => resolve());
      sock.setEncoding('utf8');
      sock.on('data', (chunk) => this._onData(chunk));
      sock.on('error', (err) => {
        reject(err);
        for (const { reject: r } of this._pending.values()) r(err);
        this._pending.clear();
      });
      sock.on('close', () => this.emit('terminated'));
      this._socket = sock;
    });
  }

  _onData(chunk) {
    this._buf += chunk;
    while (true) {
      const headerEnd = this._buf.indexOf('\r\n\r\n');
      if (headerEnd === -1) break;
      const header = this._buf.slice(0, headerEnd);
      const match = header.match(/Content-Length:\s*(\d+)/i);
      if (!match) { this._buf = this._buf.slice(headerEnd + 4); continue; }
      const len = parseInt(match[1], 10);
      const bodyStart = headerEnd + 4;
      if (this._buf.length < bodyStart + len) break;
      const body = this._buf.slice(bodyStart, bodyStart + len);
      this._buf = this._buf.slice(bodyStart + len);
      try { this._dispatch(JSON.parse(body)); } catch (_) {}
    }
  }

  _dispatch(msg) {
    if (msg.type === 'response') {
      const cb = this._pending.get(msg.request_seq);
      if (cb) {
        this._pending.delete(msg.request_seq);
        if (msg.success) cb.resolve(msg.body || {});
        else cb.reject(new Error(msg.message || 'DAP error'));
      }
    } else if (msg.type === 'event') {
      this.emit(msg.event, msg.body || {});
      if (msg.event === 'stopped') this.emit('stopped', msg.body);
      if (msg.event === 'terminated') this.emit('terminated', msg.body);
    }
  }

  send(command, args = {}) {
    return new Promise((resolve, reject) => {
      const seq = this._seq++;
      const msg = JSON.stringify({ seq, type: 'request', command, arguments: args });
      const frame = `Content-Length: ${Buffer.byteLength(msg, 'utf8')}\r\n\r\n${msg}`;
      this._pending.set(seq, { resolve, reject });
      this._socket.write(frame, 'utf8');
    });
  }

  disconnect() {
    if (this._socket) { this._socket.destroy(); this._socket = null; }
  }
}
```

写文件 `~/.claude/mcp-servers/mcp-delve/src/session.js`：

```js
import { spawn } from 'child_process';
import net from 'net';
import { DAPClient } from './dap-client.js';

const DLV = process.env.DLV_PATH || '/home/yang/go/bin/dlv';

function getFreePort() {
  return new Promise((resolve, reject) => {
    const srv = net.createServer();
    srv.listen(0, '127.0.0.1', () => {
      const { port } = srv.address();
      srv.close(() => resolve(port));
    });
    srv.on('error', reject);
  });
}

async function waitForPort(port, maxMs = 10000) {
  const deadline = Date.now() + maxMs;
  while (Date.now() < deadline) {
    try {
      await new Promise((res, rej) => {
        const s = net.createConnection({ host: '127.0.0.1', port }, res);
        s.on('error', rej);
      }).then(s => s && s.destroy && s.destroy());
      return;
    } catch {
      await new Promise(r => setTimeout(r, 200));
    }
  }
  throw new Error(`dlv dap did not listen on :${port} within ${maxMs}ms`);
}

export class DebugSession {
  constructor() {
    this._proc = null;
    this._dap = null;
    this._state = 'idle';     // idle | running | stopped | terminated
    this._threadId = 1;
    this._location = null;    // { file, line, func }
  }

  async launch({ program, mode = 'debug', args = [], env = {}, cwd, stopOnEntry = true }) {
    if (this._state !== 'idle') throw new Error('Session already active');
    const port = await getFreePort();
    this._proc = spawn(DLV, ['dap', `--listen=127.0.0.1:${port}`], {
      cwd: cwd || process.cwd(),
      env: { ...process.env, ...env },
      stdio: ['ignore', 'pipe', 'pipe'],
    });
    this._proc.on('exit', () => { this._state = 'terminated'; });

    await waitForPort(port);
    this._dap = new DAPClient();
    await this._dap.connect('127.0.0.1', port);

    this._dap.on('stopped', (body) => {
      this._state = 'stopped';
      this._threadId = body.threadId || 1;
    });
    this._dap.on('terminated', () => { this._state = 'terminated'; });

    await this._dap.send('initialize', { adapterID: 'dlv', pathFormat: 'path' });
    await this._dap.send('launch', {
      mode, program,
      args, env, cwd: cwd || process.cwd(),
      stopOnEntry,
    });
    await this._dap.send('configurationDone', {});
    this._state = 'running';

    if (stopOnEntry) await this._waitStopped();
    return { state: this._state, location: this._location };
  }

  async attach({ port }) {
    if (this._state !== 'idle') throw new Error('Session already active');
    this._dap = new DAPClient();
    await this._dap.connect('127.0.0.1', port);
    this._dap.on('stopped', (body) => { this._state = 'stopped'; this._threadId = body.threadId || 1; });
    this._dap.on('terminated', () => { this._state = 'terminated'; });
    await this._dap.send('initialize', { adapterID: 'dlv', pathFormat: 'path' });
    await this._dap.send('attach', { mode: 'remote', port });
    await this._dap.send('configurationDone', {});
    this._state = 'running';
    return { state: this._state };
  }

  _waitStopped(timeoutMs = 30000) {
    return new Promise((resolve, reject) => {
      if (this._state === 'stopped') return resolve();
      if (this._state === 'terminated') return reject(new Error('Process terminated'));
      const t = setTimeout(() => {
        this._dap.off('stopped', onStop);
        this._dap.off('terminated', onTerm);
        reject(new Error('Timeout waiting for stop'));
      }, timeoutMs);
      const onStop = async () => {
        clearTimeout(t);
        this._dap.off('terminated', onTerm);
        this._location = await this._currentLocation();
        resolve();
      };
      const onTerm = () => {
        clearTimeout(t);
        this._dap.off('stopped', onStop);
        reject(new Error('Process terminated'));
      };
      this._dap.once('stopped', onStop);
      this._dap.once('terminated', onTerm);
    });
  }

  async _currentLocation() {
    try {
      const r = await this._dap.send('stackTrace', { threadId: this._threadId, startFrame: 0, levels: 1 });
      const f = r.stackFrames?.[0];
      if (!f) return null;
      return { file: f.source?.path || '?', line: f.line, func: f.name };
    } catch { return null; }
  }

  async setBreakpoints(file, lines) {
    const r = await this._dap.send('setBreakpoints', {
      source: { path: file },
      breakpoints: lines.map(l => ({ line: l })),
    });
    return (r.breakpoints || []).map(b => ({ id: b.id, line: b.line, verified: b.verified }));
  }

  async _resumeAndWait(command, args = {}) {
    if (this._state !== 'stopped') throw new Error(`Cannot ${command}: state is ${this._state}`);
    this._state = 'running';
    await this._dap.send(command, { threadId: this._threadId, ...args });
    await this._waitStopped();
    return this._stopContext();
  }

  async continue() { return this._resumeAndWait('continue'); }
  async next()      { return this._resumeAndWait('next'); }
  async stepIn()    { return this._resumeAndWait('stepIn'); }
  async stepOut()   { return this._resumeAndWait('stepOut'); }

  async _stopContext() {
    const loc = this._location;
    let frames = [];
    try {
      const r = await this._dap.send('stackTrace', { threadId: this._threadId, startFrame: 0, levels: 10 });
      frames = (r.stackFrames || []).map(f => ({
        id: f.id, name: f.name, line: f.line, file: f.source?.path || '?',
      }));
    } catch {}
    return { state: this._state, reason: 'stopped', ...loc, frames };
  }

  async stackTrace() {
    const r = await this._dap.send('stackTrace', { threadId: this._threadId, startFrame: 0, levels: 20 });
    return (r.stackFrames || []).map(f => ({
      id: f.id, name: f.name, line: f.line, file: f.source?.path || '?',
    }));
  }

  async variables(frameId) {
    const scopes = await this._dap.send('scopes', { frameId });
    const vars = [];
    for (const scope of (scopes.scopes || []).slice(0, 2)) {
      const r = await this._dap.send('variables', { variablesReference: scope.variablesReference });
      vars.push(...(r.variables || []).map(v => ({ name: v.name, value: v.value, type: v.type, scope: scope.name })));
    }
    return vars;
  }

  async evaluate(expression, frameId) {
    const r = await this._dap.send('evaluate', { expression, frameId, context: 'repl' });
    return { result: r.result, type: r.type };
  }

  async goroutines() {
    const r = await this._dap.send('threads', {});
    return (r.threads || []).map(t => ({ id: t.id, name: t.name }));
  }

  status() {
    return { state: this._state, ...(this._location || {}) };
  }

  async stop() {
    try { await this._dap.send('disconnect', { terminateDebuggee: true }); } catch {}
    this._dap?.disconnect();
    this._proc?.kill('SIGTERM');
    this._state = 'idle';
    this._proc = null;
    this._dap = null;
    this._location = null;
  }
}
```

```js
import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
import { z } from 'zod';
import { DebugSession } from './session.js';

const session = new DebugSession();
const server = new McpServer({ name: 'mcp-delve', version: '1.0.0' });

function ok(data) { return { content: [{ type: 'text', text: JSON.stringify(data, null, 2) }] }; }
function err(e)   { return { content: [{ type: 'text', text: `ERROR: ${e.message}` }], isError: true }; }

server.tool('launch_debug',
  'Launch a Go program under delve DAP debugger',
  {
    program: z.string().describe('Path to Go source dir (mode=debug) or binary (mode=exec)'),
    mode: z.enum(['debug','exec']).default('debug'),
    args: z.array(z.string()).default([]).describe('Program arguments'),
    cwd: z.string().optional().describe('Working directory'),
    stop_on_entry: z.boolean().default(true),
  },
  async ({ program, mode, args, cwd, stop_on_entry }) => {
    try { return ok(await session.launch({ program, mode, args, cwd, stopOnEntry: stop_on_entry })); }
    catch (e) { return err(e); }
  }
);

server.tool('attach_debug',
  'Attach to an already-running dlv dap server',
  { port: z.number().describe('Port where dlv dap is listening') },
  async ({ port }) => {
    try { return ok(await session.attach({ port })); }
    catch (e) { return err(e); }
  }
);

server.tool('set_breakpoints',
  'Set breakpoints in a source file (replaces previous breakpoints in that file)',
  {
    file: z.string().describe('Absolute path to source file'),
    lines: z.array(z.number()).describe('Line numbers'),
  },
  async ({ file, lines }) => {
    try { return ok(await session.setBreakpoints(file, lines)); }
    catch (e) { return err(e); }
  }
);

server.tool('continue_exec',
  'Continue execution until next breakpoint or program exit',
  {},
  async () => {
    try { return ok(await session.continue()); }
    catch (e) { return err(e); }
  }
);

server.tool('next',
  'Step over — execute the next line without entering function calls',
  {},
  async () => {
    try { return ok(await session.next()); }
    catch (e) { return err(e); }
  }
);

server.tool('step_in',
  'Step into — enter the function call on the current line',
  {},
  async () => {
    try { return ok(await session.stepIn()); }
    catch (e) { return err(e); }
  }
);

server.tool('step_out',
  'Step out — run until returning from the current function',
  {},
  async () => {
    try { return ok(await session.stepOut()); }
    catch (e) { return err(e); }
  }
);

server.tool('get_stack',
  'Get the current goroutine stack trace (up to 20 frames)',
  {},
  async () => {
    try { return ok(await session.stackTrace()); }
    catch (e) { return err(e); }
  }
);

server.tool('get_variables',
  'Get local variables and arguments at a stack frame',
  { frame_id: z.number().describe('Frame ID from get_stack') },
  async ({ frame_id }) => {
    try { return ok(await session.variables(frame_id)); }
    catch (e) { return err(e); }
  }
);

server.tool('evaluate',
  'Evaluate an expression in the context of a stack frame',
  {
    expression: z.string().describe('Go expression to evaluate'),
    frame_id: z.number().optional().describe('Frame ID from get_stack'),
  },
  async ({ expression, frame_id }) => {
    try { return ok(await session.evaluate(expression, frame_id)); }
    catch (e) { return err(e); }
  }
);

server.tool('list_goroutines',
  'List all goroutines (reported as DAP threads)',
  {},
  async () => {
    try { return ok(await session.goroutines()); }
    catch (e) { return err(e); }
  }
);

server.tool('get_status',
  'Get current debug session state: idle | running | stopped | terminated',
  {},
  async () => { return ok(session.status()); }
);

server.tool('stop_debug',
  'Terminate the debug session and kill the debuggee process',
  {},
  async () => {
    try { await session.stop(); return ok({ stopped: true }); }
    catch (e) { return err(e); }
  }
);

const transport = new StdioServerTransport();
await server.connect(transport);
```
